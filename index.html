function doPost(e) {
  try {
    var data = JSON.parse(e.postData.contents);
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    writeRecords(ss, data.records || []);
    writeStores(ss, data.stores || []);
    writeLog(ss, data.backedUpAt);
  } catch(err) {
    Logger.log(err);
  }
}

function writeRecords(ss, records) {
  var sheet = ss.getSheetByName('Records') || ss.insertSheet('Records');
  sheet.clearContents();
  sheet.appendRow(['#','Store','Period','Score Obtained','Max Score','Percentage','Zone','Saved At']);
  records.forEach(function(r, i) {
    var zone = r.pct >= 90 ? 'Green' : r.pct >= 80 ? 'Yellow' : 'Red';
    sheet.appendRow([i+1, r.store, r.period, r.tO, r.tM, r.pct+'%', zone, r.savedAt]);
  });
}

function writeStores(ss, stores) {
  var sheet = ss.getSheetByName('Stores') || ss.insertSheet('Stores');
  sheet.clearContents();
  sheet.appendRow(['Store Name']);
  stores.forEach(function(s) { sheet.appendRow([s]); });
}

function writeLog(ss, ts) {
  var sheet = ss.getSheetByName('BackupLog') || ss.insertSheet('BackupLog');
  sheet.appendRow([new Date(), ts, 'Backup completed']);
}
